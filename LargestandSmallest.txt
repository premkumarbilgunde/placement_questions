public class Largestandsmallest {

	public static void main(String[] args) {
		int num=98271;
		int Largest=0;
		int Smallest=9;
		
		int temp=num;
		
		while(temp>0) {
			int digit=temp%10;
			if(digit>Largest) {
				Largest=digit;
			}
			if(digit<Smallest) {
				Smallest=digit;
			}
			temp=temp/10;
			
		}
		System.out.println("Largest digit"+ Largest);
		System.out.println("Smallest digit"+ Smallest);

		
	}

}
